import requests from bs4 import BeautifulSoup # Make a request to the website url = 'https://discord.com/' response = requests.get(url) # Parse the HTML content soup = BeautifulSoup(response.content, 'html.parser') # Extract specific content (for example, all paragraphs) paragraphs = soup.find_all('p') # Print or save the extracted content for p in paragraphs: print(p.text)
